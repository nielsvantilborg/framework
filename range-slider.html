Range slider

Notes

  • Class 'noUi-extended' let the handles stay within the slide bar (instead of the default overlap)
  • UniqueId is a variable name

Source

  • http://refreshless.com/nouislider/

Range slider one handle

Javascript

<script type='text/javascript' src='js/range-slider-8.5.1.js'></script> <script type="text/javascript"> // Range slider one handle var rangeSlider = document.getElementById('rangeSliderUniqueId1'); noUiSlider.create(rangeSlider, { start: [ 4000 ], range: { 'min': [ 2000 ], 'max': [ 10000 ] } }); var rangeSliderValueElement = document.getElementById('rangeSliderUniqueIdValue1'); rangeSlider.noUiSlider.on('update', function( values, handle ) { rangeSliderValueElement.innerHTML = values[handle]; }); </script>

HTML

<div class="rangeSliderContainer"> <div id="rangeSliderUniqueId1" class="noUi-extended"></div> <span id="rangeSliderUniqueIdValue1" class="rangeSliderValue"></span> </div>

CSS

Range slider two handles

Javascript

<script type='text/javascript' src='js/range-slider-8.5.1.js'></script> <script type="text/javascript"> var snapSlider = document.getElementById('rangeSliderUniqueId2'); noUiSlider.create(snapSlider, { start: [ 0, 1000 ], step: 10, connect: true, range: { 'min': 0, 'max': 1000 } }); var snapValues = [ document.getElementById('rangeSliderUniqueIdValueMin2'), document.getElementById('rangeSliderUniqueIdValueMax2') ]; snapSlider.noUiSlider.on('update', function( values, handle ) { snapValues[handle].innerHTML = values[handle]; }); </script>

HTML

<div class="rangeSliderContainer"> <div id="rangeSliderUniqueId2" class="noUi-extended"></div> <span id="rangeSliderUniqueIdValueMin2" class="rangeSliderValueMin"></span> <span id="rangeSliderUniqueIdValueMax2" class="rangeSliderValueMax"></span> </div>

CSS
